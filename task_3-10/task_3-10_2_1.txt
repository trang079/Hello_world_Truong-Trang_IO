#!/bin/bash
# Цикл от 1 до 20
for i in {1..20}; do
  # Если число равно 15, прерываем цикл
  if [ $i -eq 15 ]; then
    echo "Число 15. Выход из цикла."
    break
  fi
  # Если число четное, пропускаем итерацию
  if [ $((i % 2)) -eq 0 ]; then
    continue
  fi
  # Вывод нечетного числа
  echo "Нечетное число: $i"
done
