#!/bin/bash

# Вывод заголовка таблицы
printf "%-15s %-7s %-7s %-7s %-7s\n" "Файл" "A" "T" "G" "C"

# Цикл по всем *.fasta файлам
for file in *.fasta; do

  # Пропускаем пустые файлы
  if [ ! -s "$file" ]; then
    continue
  fi

  # Обработка последовательности (удаление заголовка и переносов)
  sequence=$(grep -v ">" "$file" | tr -d '\n')

  # Подсчет нуклеотидов
  A=$(echo "$sequence" | grep -o "A" | wc -l)
  T=$(echo "$sequence" | grep -o "T" | wc -l)
  G=$(echo "$sequence" | grep -o "G" | wc -l)
  C=$(echo "$sequence" | grep -o "C" | wc -l)

  # Вывод результата в таблицу
  printf "%-15s %-7s %-7s %-7s %-7s\n" "$file" "$A" "$T" "$G" "$C"

done
