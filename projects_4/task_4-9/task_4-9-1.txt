# Программа для поиска максимального из двух чисел (X и Y)

# 1. Ввод данных
x = float(input("Введите число X: "))
y = float(input("Введите число Y: "))

# 2. Проверка условия
if x > y:
    max_value = x
else:
    max_value = y

# 3. Вывод результата
print("Максимальное число:", max_value)
