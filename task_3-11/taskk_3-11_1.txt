#!/bin/bash

# Функция для проверки прав суперпользователя
check_root() {
  # Переменная $EUID хранит ID текущего пользователя. У root он всегда равен 0.
  if [ "$EUID" -ne 0 ]; then
    echo "Ошибка: Этот скрипт должен быть запущен от имени root!"
    exit 1
  fi
}

# Вызов функции
check_root

# Если мы дошли до этой строки, значит пользователь — root
echo "Проверка пройдена: запуск от имени суперпользователя."
