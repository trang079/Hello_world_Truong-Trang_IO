# Исходный массив
array = [5, 3, 8, 4, 6]
N = len(array)

# Алгоритм пузырьковой сортировки
for i in range(N):
    # Внутренний цикл: от 0 до N-2
    for j in range(N - 1):
        # Условие проверки: a[j] > a[j+1]
        if array[j] > array[j + 1]:
            # Обмен значениями (перестановка)
            temp = array[j]
            array[j] = array[j + 1]
            array[j + 1] = temp

print("Отсортированный массив:", array)
